비밀번호 

    이벤트 처리를 목적으로 하는 함수
    bind() : $(선택자).bind(이벤트, function(){})
        $(선택자).bind({이벤트1:function(){},
                        이벤트2:function(){},
                        ...})
    one() : 일회성 처리가 필요시, bind와 사용법 동일
    on() : 동적으로 추가할 요소에 대한 이벤트를 미리 정의해 놓는 기능
            이벤트가 정의되는 시점에는 대상이 존재하지 않음, 이 이벤트는 document 객체에 설정
            $(document).on('이벤트','선택자',function(){})
